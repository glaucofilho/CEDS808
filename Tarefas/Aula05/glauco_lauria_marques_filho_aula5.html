1.a.I SVM

Nesse item iremos aplicar o algoritmo de SVM visto em aula.

Faremos uma modelagem primeiro variando os kernels do SVM

In[]:
SVM_kernels = ["linear", "poly", "rbf", "sigmoid"]
SVM_mean_accuracy = []
for kernel in SVM_kernels:
    print(f"Kernel = {kernel}")
    svm_classifier = SVC(kernel=kernel)
    acuracias = []
    kf = KFold(n_splits=5)
    for train_index, test_index in kf.split(X):
        X_train , X_test = X.iloc[train_index], X.iloc[test_index]
        y_train , y_test = y.iloc[train_index], y.iloc[test_index]
        svm_classifier.fit(X_train,np.ravel(y_train))
        accuracy = svm_classifier.score(X_test, np.ravel(y_test))
        acuracias.append(accuracy)
    acuracia_media = np.mean(acuracias)
    SVM_mean_accuracy.append(acuracia_media)
    print(f"Mean Accuracy at 5 folds = {acuracia_media}\n")

Nesse item iremos aplicar o algoritmo de Perceptron visto em aula. Variando os parametros de penalidade.

In[]:
# Convertendo os rtulos para nmeros (necessrio para Perceptron no sklearn)
le = LabelEncoder()
y_num = le.fit_transform(y)
y_num = pd.DataFrame({"class":y_num})
y_num = y_num["class"]


# Inicializando a validao cruzada k-fold
kf = KFold(n_splits=5)

fold_number = 1
accuracies = []

perceptron_penalities_mean_accuracy = []

penalities = [None, "l2", "l1", "elasticnet"]
penalities_str = ["None", "l2", "l1", "elasticnet"]

for penality in penalities:
    if penality is not None:
        print(f"Penalidade: {penality}")
    else:
        print("Penalidade: None")
    # Iterando sobre os folds
    for train_index, test_index in kf.split(X):
        # Dividindo os dados em conjunto de treinamento e teste
        X_train, X_test = X.iloc[train_index], X.iloc[test_index]
        y_train, y_test = y_num.iloc[train_index], y_num.iloc[test_index]

        # Criando e treinando o modelo Perceptron
        perceptron = Perceptron(penalty=penality)
        perceptron.fit(X_train, np.ravel(y_train))

        # Fazendo previses no conjunto de teste
        y_pred = perceptron.predict(X_test)

        # Calculando a acurcia do fold atual
        accuracy = np.mean(y_pred == np.ravel(y_test))
        accuracies.append(accuracy)

        # Imprimindo o resultado do fold atual
        print(f"Fold {fold_number}: Accuracy = {accuracy}")
        fold_number += 1
    fold_number = 1

    # Calculando a mdia das acurcias
    mean_accuracy = np.mean(accuracies)
    perceptron_penalities_mean_accuracy.append(mean_accuracy)
    print(f'Mean Accuracy of all Folds: {mean_accuracy}\n')

In[]:
class Perceptron:
    def __init__(self, eta=0.5, n_iter=10, limiar=-0.5):
        self.eta = eta
        self.n_iter = n_iter
        self.limiar = limiar

    def fit(self, X, y):
        self.w_ = np.zeros(1 + X.shape[1])
        # Definindo o erro do limiar em w0, como no slide
        self.w_[0] = - self.limiar
        self.errors_ = []

        for _ in range(self.n_iter):
            # erros da geracao
            errors = 0

            for xi, target in zip(X, y):
                
                # Inserindo o valor 1 para o w0
                xi = np.insert(xi, 0, 1)
                # Soma ponderada pelos pesos
                weighted_sum = np.dot(xi, self.w_)
                # Ativacao do perceptron
                output = self.ativacao(weighted_sum + self.limiar)
                # erro da previsao
                error = target - output
                # Atualizacao dos pesos
                self.w_ += self.eta * error * xi
    
                errors += int(error != 0.0)
            # Adicionando o erro da geracao
            self.errors_.append(errors)

    def ativacao(self, x):
        """Funo de ativao que aplica o limiar."""
        return 1 if x >= 0 else 0  # Ajuste conforme a definio do limiar

    def predict(self, X):
        """Previso da classe usando o vetor de pesos e o limiar."""
        X = np.hstack([np.ones((X.shape[0], 1)), X])
        weighted_sum = np.dot(X, self.w_)
        return np.where(weighted_sum >= 0, 1, 0)

Definicoes:

Febre, 1 para Sim, 0 para Nao.

Enjoo, 1 para Sim, 0 para Nao.

Manchas, 1 para Grandes, 0 para Pequenas.

Dores, 1 para Sim, 0 para nao.

Diagnostico, 1 para Doente, 0 para Saudavel.
